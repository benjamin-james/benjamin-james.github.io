#+BEGIN_SRC emacs-lisp :exports results :results none
    (defun cv-bold-self (rendered-name name _)
    "Make RENDERED-NAME bold when NAME is mine.
  Leave RENDERED-NAME unchanged otherwise. Return RENDERED-NAME."
    (let ((given (cdr (assq 'given name)))
          (family (cdr (assq 'family name))))
      (when (and (member given '("Benjamin T." "Benjamin" "Ben" "Benjamin T" "B." "Benjamin Thomas"))
  	       (member family '("James" "James*")))
        (push '(font-weight . "bold") (car rendered-name)))
      rendered-name))
#+END_SRC
#+bind: citeproc-name-postprocess-functions (cv-bold-self)
